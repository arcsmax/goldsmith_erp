> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make Goldsmith ERP reliable enough for daily workshop use on a single self-hosted server with automated backups, system monitoring, and security hardening.

```python
#!/usr/bin/env python3
"""Create an admin user for Goldsmith ERP.

Usage: poetry run python scripts/create-admin.py --email admin@example.com --password secret
"""
import asyncio
import argparse
import sys
import os

# Add src to path
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..', 'src'))

from goldsmith_erp.db.session import AsyncSessionLocal
from goldsmith_erp.db.models import User, UserRole
from goldsmith_erp.core.security import get_password_hash
from sqlalchemy import select


async def create_admin(email: str, password: str, first_name: str = "Admin", last_name: str = "User"):
    async with AsyncSessionLocal() as db:
        # Check if user exists
        result = await db.execute(select(User).where(User.email == email))
        existing = result.scalar_one_or_none()
        if existing:
            print(f"Benutzer {email} existiert bereits. Überspringe.")
            return

        user = User(
            email=email,
            hashed_password=get_password_hash(password),
            first_name=first_name,
            last_name=last_name,
            role=UserRole.ADMIN,
            is_active=True,
        )
        db.add(user)
        await db.commit()
        print(f"Admin-Benutzer erstellt: {email}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Goldsmith ERP Admin erstellen")
    parser.add_argument("--email", required=True, help="E-Mail-Adresse")
    parser.add_argument("--password", required=True, help="Passwort")
    parser.add_argument("--first-name", default="Admin", help="Vorname")
    parser.add_argument("--last-name", default="User", help="Nachname")
    args = parser.parse_args()
    asyncio.run(create_admin(args.email, args.password, args.first_name, args.last_name))
```

- [ ] **Step 1: Create backup script**

`scripts/backup.sh` — runs `pg_dump` against the Podman PostgreSQL container, compresses with gzip, verifies integrity, manages retention (7 daily + 4 weekly + 3 monthly). Reads `BACKUP_DIR` and `BACKUP_CLOUD_URL` from `.env.production`. Publishes notification via backend API on success/failure.

Key logic:
```bash
BACKUP_FILE="${BACKUP_DIR}/goldsmith_erp_$(date +%Y-%m-%d_%H%M%S).sql.gz"
podman-compose -f podman-compose.prod.yml exec -T db pg_dump -U "$POSTGRES_USER" "$POSTGRES_DB" | gzip > "$BACKUP_FILE"
# Verify integrity
gzip -t "$BACKUP_FILE" || { echo "FEHLER: Backup-Datei beschädigt!"; exit 1; }
# Run cloud sync if configured
[ -n "$BACKUP_CLOUD_URL" ] && bash scripts/backup-sync.sh "$BACKUP_FILE"
# Clean old backups (retention policy)
# Notify admin via backend API (uses curl to hit the local notification endpoint)
BACKUP_SIZE=$(du -h "$BACKUP_FILE" | cut -f1)
curl -s -X POST "http://localhost:8000/api/v1/admin/notify-backup" \
  -H "Content-Type: application/json" \
  -d "{\"status\": \"success\", \"filename\": \"$(basename $BACKUP_FILE)\", \"size\": \"$BACKUP_SIZE\"}" \
  || echo "WARNUNG: Backup-Benachrichtigung konnte nicht gesendet werden."
```

- [ ] **Step 1: Add retry with exponential backoff to publish_event**

Replace the current `publish_event` in `pubsub.py` with a version that retries up to 3 times with exponential backoff (0.5s, 1s, 2s). On final failure, log the error but do NOT raise — publishing is best-effort.

```python
async def publish_event(channel: str, message: str) -> None:
    """Publish a message to Redis. Retries with backoff. Never raises."""
    for attempt in range(3):
        try:
            async with get_redis_client() as client:
                await client.publish(channel, message)
                return
        except Exception as e:
            if attempt < 2:
                await asyncio.sleep(0.5 * (2 ** attempt))
                logger.warning(f"Redis publish retry {attempt + 1}/3: {e}")
            else:
                logger.error(f"Redis publish failed after 3 attempts: {e}")
```

```python
"""System health monitoring service.

Checks database, Redis, and disk health. Provides system metrics
for the admin dashboard. Creates notifications for critical issues.
"""
import logging
import os
import time
import shutil
from datetime import datetime
from typing import Optional
from pathlib import Path

from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import text, func, select

from goldsmith_erp.core.config import settings
from goldsmith_erp.db.models import Order, OrderStatusEnum

logger = logging.getLogger(__name__)

# Track app start time for uptime calculation
_app_start_time = time.time()


class SystemHealthService:
    """Provides system health checks and metrics."""

    @staticmethod
    async def check_database(db: AsyncSession) -> dict:
        """Check database connectivity and measure latency."""
        start = time.time()
        try:
            await db.execute(text("SELECT 1"))
            latency = round((time.time() - start) * 1000, 1)
            return {"status": "up", "latency_ms": latency}
        except Exception as e:
            logger.error(f"Database health check failed: {e}")
            return {"status": "down", "error": str(e)}

    @staticmethod
    async def check_redis() -> dict:
        """Check Redis connectivity and measure latency."""
        start = time.time()
        try:
            from goldsmith_erp.core.pubsub import get_redis_client
            async with get_redis_client() as client:
                await client.ping()
                latency = round((time.time() - start) * 1000, 1)
                info = await client.info("memory")
                return {
                    "status": "up",
                    "latency_ms": latency,
                    "used_memory_mb": round(info.get("used_memory", 0) / 1024 / 1024, 1),
                }
        except Exception as e:
            logger.error(f"Redis health check failed: {e}")
            return {"status": "down", "error": str(e)}

    @staticmethod
    def check_disk() -> dict:
        """Check disk space on the data volume."""
        try:
            usage = shutil.disk_usage("/")
            free_gb = round(usage.free / (1024**3), 1)
            total_gb = round(usage.total / (1024**3), 1)
            used_pct = round((usage.used / usage.total) * 100, 1)
            status = "ok" if used_pct < 80 else ("warning" if used_pct < 95 else "critical")
            return {
                "status": status,
                "free_gb": free_gb,
                "total_gb": total_gb,
                "used_percent": used_pct,
            }
        except Exception as e:
            return {"status": "unknown", "error": str(e)}

    @staticmethod
    def get_uptime() -> int:
        """Return seconds since application start."""
        return int(time.time() - _app_start_time)

    @staticmethod
    async def get_full_health(db: AsyncSession) -> dict:
        """Combined health check for all components."""
        db_health = await SystemHealthService.check_database(db)
        redis_health = await SystemHealthService.check_redis()
        disk_health = SystemHealthService.check_disk()

        components = {
            "database": db_health,
            "redis": redis_health,
            "disk": disk_health,
        }

        # Overall status: worst component wins
        statuses = [c.get("status", "unknown") for c in components.values()]
        if "down" in statuses or "critical" in statuses:
            overall = "unhealthy"
        elif "warning" in statuses:
            overall = "degraded"
        else:
            overall = "healthy"

        return {
            "status": overall,
            "components": components,
            "version": settings.APP_VERSION,
            "uptime_seconds": SystemHealthService.get_uptime(),
        }

    @staticmethod
    async def get_business_metrics(db: AsyncSession) -> dict:
        """Get business metrics from database (survives restart)."""
        now = datetime.utcnow()
        month_start = now.replace(day=1, hour=0, minute=0, second=0, microsecond=0)

        # Orders this month
        result = await db.execute(
            select(func.count(Order.id)).where(Order.created_at >= month_start)
        )
        orders_this_month = result.scalar() or 0

        # Completed this month
        result = await db.execute(
            select(func.count(Order.id)).where(
                Order.status == OrderStatusEnum.COMPLETED,
                Order.completed_at >= month_start,
            )
        )
        completed_this_month = result.scalar() or 0

        return {
            "orders_this_month": orders_this_month,
            "completed_this_month": completed_this_month,
        }

    @staticmethod
    def get_last_backup_info() -> Optional[dict]:
        """Read last backup status from log file."""
        backup_dir = Path(settings.BACKUP_DIR).expanduser()
        if not backup_dir.exists():
            return None

        backups = sorted(backup_dir.glob("goldsmith_erp_*.sql.gz"), reverse=True)
        if not backups:
            return None

        latest = backups[0]
        return {
            "filename": latest.name,
            "size_mb": round(latest.stat().st_size / (1024 * 1024), 1),
            "timestamp": datetime.fromtimestamp(latest.stat().st_mtime).isoformat(),
            "backup_count": len(backups),
        }
```

```bash
#!/bin/bash
# Rotate SECRET_KEY in .env.production
# WARNING: This invalidates all active JWT tokens — users must re-login
set -e

ENV_FILE=".env.production"
if [ ! -f "$ENV_FILE" ]; then
    echo "FEHLER: $ENV_FILE nicht gefunden. Führe zuerst 'make setup' aus."
    exit 1
fi

echo "WARNUNG: Alle aktiven Sitzungen werden ungültig."
echo "Benutzer müssen sich erneut anmelden."
read -p "Fortfahren? (j/n) " -n 1 -r
echo
if [[ ! $REPLY =~ ^[jJyY]$ ]]; then
    echo "Abgebrochen."
    exit 0
fi

# Generate new SECRET_KEY
NEW_KEY=$(python3 -c "import secrets; print(secrets.token_urlsafe(64))")
# Replace in .env.production
sed -i.bak "s|^SECRET_KEY=.*|SECRET_KEY=${NEW_KEY}|" "$ENV_FILE"
rm -f "${ENV_FILE}.bak"

echo "SECRET_KEY rotiert. Starte Backend neu..."
podman-compose -f podman-compose.prod.yml restart backend
echo "Fertig. Alle Benutzer müssen sich erneut anmelden."
```
